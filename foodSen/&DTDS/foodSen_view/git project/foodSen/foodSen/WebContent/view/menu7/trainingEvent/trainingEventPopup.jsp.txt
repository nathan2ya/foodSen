	$(document).ready(function(){
		
		//이미지 시작지점을 바꿈
		var cnt=$('#cnt').val();
		
		$('#prev').click(function(){
			var offset=$('#img').offset();
	
			if(offset.left < 8){
				offset.left += 350;
			}
			
			$('#img').animate(function(){
				left: '+= 350';
			}, 400);
			
			$('#img').offset(offset);
		});
		
		$('#next').click(function(){
			var offset=$('#img').offset();
			//alert(offset.left);
			if(offset.left > -(356*(cnt-1))+60){
				offset.left -= 350;
			}
	
			$('#img').animate(function(){
				left: '-= 350';
			},400);
			
			$('#img').offset(offset);
		});
	});
	
</script>
